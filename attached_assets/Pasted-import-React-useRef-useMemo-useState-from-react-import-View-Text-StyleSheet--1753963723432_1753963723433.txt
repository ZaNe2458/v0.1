  Text,
  StyleSheet,
  Dimensions,
  Image,

const carTypes = [
  { id: 'small', name: '–ñ–∏–∂–∏–≥ —Ç—ç—Ä—ç–≥', img: require('../assets/images/sedan.png') },
  { id: 'medium', name: '–î—É–Ω–¥ –æ–≤—Ä—ã–Ω', img: require('../assets/images/sav.webp') },
  { id: 'jeep', name: '–ñ–∏–π–ø', img: require('../assets/images/jeep.png') },
  { id: 'bigJeep', name: '–¢–æ–º –æ–≤—Ä—ã–Ω –∂–∏–π–ø', img: require('../assets/images/bigsuv.png') },
  { id: 'van', name: '–¢—Ä–∞–Ω–∑–∏—Ç', img: require('../assets/images/van.png') },
  { id: 'moto', name: '–ú–æ—Ç–æ—Ü–∏–∫–ª', img: require('../assets/images/moto.png') },
];

export default function HomeScreen() {
  const bottomSheetModalRef = useRef<BottomSheetModal>(null);
  const snapPoints = useMemo(() => ['60%'], []);
  const [selectedCarWash, setSelectedCarWash] = useState<any>(null);
  const [activeCarType, setActiveCarType] = useState<string>('small');

  const handleMarkerPress = (location: any) => {
    setSelectedCarWash(location);
    bottomSheetModalRef.current?.present();
  };

  return (
    <BottomSheetModalProvider>
      <View style={styles.container}>
        <MapView
          style={styles.map}
          initialRegion={{
            latitude: 47.9185,
            longitude: 106.917,
            latitudeDelta: 0.02,
            longitudeDelta: 0.02,
          }}
        >
          {carWashLocations.map((loc) => (
            <Marker
              key={loc.id}
              coordinate={{ latitude: loc.latitude, longitude: loc.longitude }}
              title={loc.name}
              onPress={() => handleMarkerPress(loc)}
            />
          ))}
        </MapView>

        <BottomSheetModal ref={bottomSheetModalRef} index={0} snapPoints={snapPoints}>
          <ScrollView style={styles.sheetContent} contentContainerStyle={{ paddingBottom: 40 }}>
            {selectedCarWash && (
              <>
                {/* Header */}
                <View style={styles.header}>
                  <View style={{ flex: 1 }}>
                    <Text style={styles.title}>{selectedCarWash.name}</Text>
                    <Text>üìû {selectedCarWash.phone}</Text>
                    <Text>üìß {selectedCarWash.email}</Text>
                    <Text>üïó {selectedCarWash.hours}</Text>
                    <Text>üìç {selectedCarWash.address}</Text>
                  </View>
                  <Image
                    source={require('../assets/images/logo_n.webp')}
                    style={styles.logo}
                    resizeMode="contain"
                  />
                </View>

                {/* Car types selection */}
                {carTypes.map((type) => (
                  <TouchableOpacity
                    key={type.id}
                    style={[
                      styles.carTypeBox,
                      activeCarType === type.id && styles.activeBox,
                    ]}
                    onPress={() => setActiveCarType(type.id)}
                  >
                    <Image source={type.img} style={styles.carIcon} />
                    <Text style={styles.carTypeText}>{type.name}</Text>
                    <Text style={styles.arrow}>{'>'}</Text>
                  </TouchableOpacity>
                ))}

                {/* Image gallery */}
                {selectedCarWash.images && selectedCarWash.images.length > 0 && (
                  <ScrollView
                    horizontal
                    showsHorizontalScrollIndicator={false}
                    style={styles.imageGallery}
                    contentContainerStyle={{ paddingRight: 20 }}
                  >
                    {selectedCarWash.images.map((uri: string, index: number) => (
                      <Image
                        key={index}
                        source={{ uri }}
                        style={styles.galleryImage}
                        resizeMode="cover"
                      />
                    ))}
                  </ScrollView>
                )}

                {/* Bottom large image */}
                <Image
                  source={require('../assets/images/car_wash.jpg')}
                  style={styles.bottomImage}
                  resizeMode="cover"
                />
              </>
            )}
          </ScrollView>
        </BottomSheetModal>
      </View>
    </BottomSheetModalProvider>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
  },
  map: {
    width: Dimensions.get('window').width,
    height: Dimensions.get('window').height,
  },
  sheetContent: {
    padding: 20,
  },
  header: {
    flexDirection: 'row',

    marginBottom: 10,
  },
  activeBox: {
    backgroundColor: '#b9ff3c',
  },
  carIcon: {
    width: 40,
    height: 40,
    marginRight: 12,
  },
  carTypeText: {

  },

  },
